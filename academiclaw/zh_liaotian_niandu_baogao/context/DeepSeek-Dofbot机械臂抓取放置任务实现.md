import time
import numpy as np
import rospy
from dofbot_real import RealEnv

# --- Motion Sequence Configuration ---

# Key points for the robot arm's joints (in degrees).
POINTS = [
    np.asarray([90., 90., 90., 90., 90., 90.]),
    np.asarray([130., 45.0, 53.0, 1.0, 86.0, 90.0]),
    np.asarray([136., 40.0, 53.0, 1.0, 86.0, 90.0]),
    np.asarray([130., 70.0, 53.0, 1.0, 86.0, 90.0]),
    np.asarray([180., 142.0, 70.0, 53.0, 1.0, 86.0]),
    np.asarray([180., 147.0, 40.0, 53.0, 1.0, 86.0]),
    np.asarray([180., 142.0, 40.0, 57.0, 10.0, 86.0]),
    np.asarray([180., 147.0, 70.0, 70.0, 1.0, 86.0])
]

# Number of interpolation steps between each key point.
SPLIT = [30, 10, 30, 40, 10, 10, 10]

# Gripper commands for each transition.
# A value >= 0 indicates a gripper action; -1 indicates an arm movement.
GRIPPER = [-1, -1, 130.0, -1, -1, -1, 10.0, -1]



def linear_interpolation(start, end, num_steps):
    return np.linspace(start, end, num=num_steps)


# --- Core Logic ---

def run_motion_sequence(env, points, split, gripper):
    """
    Executes the full motion sequence on the robot.

    Args:
        env (RealEnv): The robot environment instance.
        points (list): A list of normalized target points (np.ndarray).
        split (list): A list of integers for interpolation steps.
        gripper (list): A list of floats for gripper commands.
    """
    env.reset()
    num_transitions = len(points) - 1

    for t in range(num_transitions):
        gripper_command = gripper[t + 1]
        num_steps = split[t]

        if gripper_command < 0:
            # Arm movement
            start_point = points[t]
            end_point = points[t + 1]
            path = linear_interpolation(start_point, end_point, num_steps)
            print(f"from {start_point} to {end_point}")
            for p in path:
                env.step(p)
        else:
            # Gripper action
            current_gripper_pos = env.get_state()[-1]
            path = linear_interpolation(current_gripper_pos, gripper_command, num_steps)
            print(f"gripper to {gripper_command}")
            for p in path:
                env.step(gripper=p)
        
        time.sleep(0.5) # Pause between transitions


# --- Main Execution ---

def main():
    try:
        env = RealEnv()
        
        print("Starting...")
        run_motion_sequence(env, POINTS, SPLIT, GRIPPER)
        print("Sequence completed.")

    except rospy.ROSInterruptException:
        print("ROS node interrupted.")
    except Exception as e:
        print(f"An error occurred: {e}")


if __name__ == '__main__':
    main()

def evaluate(
    model_path: str,
    make_env: Callable,
    env_id: str,
    eval_episodes: int,
    run_name: str,
    Model: torch.nn.Module,
    device: torch.device = torch.device("cpu"),
    epsilon: float = 0.05,
    capture_video: bool = True,
):
    envs = gym.vector.SyncVectorEnv([make_env(env_id, 0, 0, capture_video, run_name)])
    model = Model(envs).to(device)
    model.load_state_dict(torch.load(model_path, map_location=device))
    model.eval()

    obs, _ = envs.reset()
    episodic_returns = []
    step=0
    while len(episodic_returns) < eval_episodes:
        print(step)
        if random.random() < epsilon:
            actions = np.array([envs.single_action_space.sample() for _ in range(envs.num_envs)])
        else:
            q_values = model(torch.Tensor(obs).to(device))
            actions = torch.argmax(q_values, dim=1).cpu().numpy()
        print("action:", actions)
        next_obs, _, _, _, infos = envs.step(actions)
        if "final_info" in infos:
            for info in infos["final_info"]:
                if "episode" not in info:
                    continue
                print(f"eval_episode={len(episodic_returns)}, episodic_return={info['episode']['r']}")
                episodic_returns += [info["episode"]["r"]]
        obs = next_obs
        step += 1

    return episodic_returns
